```yaml
  push:
    branches: [main]
  workflow_dispatch:
permissions:
  contents: read
  pages: write
  id-token: write
concurrency:
  group: "pages"
  cancel-in-progress: false
jobs:
  deploy:
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/configure-pages@v5
      - uses: actions/upload-pages-artifact@v3
        with:
          path: '.'
      - id: deployment
        uses: actions/deploy-pages@v5
```

  <action>
This step requires manual login to the Cloudflare dashboard. It cannot be automated.

1. Log in to https://dash.cloudflare.com
2. Select the `hamtools.cc` domain
3. Go to DNS > Records
4. Add 4 A records (do NOT modify the existing `cat-ftx1` CNAME record):

| Type | Name | Content | Proxy status |
|------|------|---------|--------------|
| A | @ | 185.199.108.153 | DNS only (grey cloud) |
| A | @ | 185.199.109.153 | DNS only (grey cloud) |
| A | @ | 185.199.110.153 | DNS only (grey cloud) |
| A | @ | 185.199.111.153 | DNS only (grey cloud) |

CRITICAL: Set proxy status to "DNS only" (grey cloud icon), NOT "Proxied" (orange cloud). Cloudflare proxy breaks GitHub Pages HTTPS cert provisioning.

CRITICAL: Do NOT modify or delete the existing `cat-ftx1` CNAME record pointing to Azure.

5. After adding records, go to GitHub repo Settings > Pages > Custom domain
6. Enter `hamtools.cc` and click Save
7. Wait for DNS check to pass (may take 5-15 minutes)
8. Ensure "Enforce HTTPS" is checked once the cert is provisioned (may take up to 1 hour)
